<script>

const canvas = document.getElementById("canvas");
const ctx = canvas.getContext("2d");

let width, height;

function resize() {
  width = canvas.width = window.innerWidth;
  height = canvas.height = window.innerHeight;
}
window.addEventListener("resize", resize);
resize();

const BOID_COUNT = 3000;
const boids = [];

let time = 0;
let frame = 0;

const entrySide = Math.floor(Math.random() * 4);

const perception = 40;
const cellSize = perception;
let grid = {};

function buildGrid() {
  grid = {};
  for (let i = 0; i < boids.length; i++) {
    const b = boids[i];
    const cx = Math.floor(b.x / cellSize);
    const cy = Math.floor(b.y / cellSize);
    const key = cx + "," + cy;
    if (!grid[key]) grid[key] = [];
    grid[key].push(b);
  }
}

function getNearbyBoids(boid) {
  const cx = Math.floor(boid.x / cellSize);
  const cy = Math.floor(boid.y / cellSize);
  let neighbors = [];

  for (let x = -1; x <= 1; x++) {
    for (let y = -1; y <= 1; y++) {
      const key = (cx + x) + "," + (cy + y);
      if (grid[key]) neighbors = neighbors.concat(grid[key]);
    }
  }
  return neighbors;
}

const fields = [];
const MAX_FIELDS = 3;

function spawnField() {
  fields.push({
    x: Math.random() * width,
    y: Math.random() * height,
    radius: 250 + Math.random() * 400,
    strength: (Math.random() > 0.5 ? 1 : -1) * (0.08 + Math.random() * 0.5),
    life: 500 + Math.random() * 400
  });
}

function updateFields() {
  if (fields.length < MAX_FIELDS && Math.random() < 0.01) spawnField();
  for (let i = fields.length - 1; i >= 0; i--) {
    fields[i].life--;
    if (fields[i].life <= 0) fields.splice(i, 1);
  }
}

class Boid {
  constructor() {
    const clusterRadius = 260;
    const angle = Math.random() * Math.PI * 2;
    const r = Math.random() * clusterRadius;
    const far = 600;

    if (entrySide === 0) {
      this.x = -far + Math.cos(angle) * r;
      this.y = height / 2 + Math.sin(angle) * r;
      this.vx = 3.5;
      this.vy = (Math.random() - 0.5) * 0.4;
    }
    if (entrySide === 1) {
      this.x = width + far + Math.cos(angle) * r;
      this.y = height / 2 + Math.sin(angle) * r;
      this.vx = -3.5;
      this.vy = (Math.random() - 0.5) * 0.4;
    }
    if (entrySide === 2) {
      this.x = width / 2 + Math.cos(angle) * r;
      this.y = -far + Math.sin(angle) * r;
      this.vx = (Math.random() - 0.5) * 0.4;
      this.vy = 3.5;
    }
    if (entrySide === 3) {
      this.x = width / 2 + Math.cos(angle) * r;
      this.y = height + far + Math.sin(angle) * r;
      this.vx = (Math.random() - 0.5) * 0.4;
      this.vy = -3.5;
    }
  }

  update() {
    let alignX = 0, alignY = 0;
    let cohesionX = 0, cohesionY = 0;
    let separationX = 0, separationY = 0;
    let count = 0;

    const perceptionSq = perception * perception;
    const neighbors = getNearbyBoids(this);

    for (let other of neighbors) {
      if (other === this) continue;

      const dx = other.x - this.x;
      const dy = other.y - this.y;
      const distSq = dx * dx + dy * dy;

      if (distSq < perceptionSq) {
        alignX += other.vx;
        alignY += other.vy;

        cohesionX += other.x;
        cohesionY += other.y;

        if (distSq < 60) {
          separationX -= dx;
          separationY -= dy;
        }
        count++;
      }
    }

    if (count > 0) {
      alignX /= count;
      alignY /= count;
      cohesionX /= count;
      cohesionY /= count;

      this.vx += (alignX - this.vx) * 0.05;
      this.vy += (alignY - this.vy) * 0.05;
      this.vx += (cohesionX - this.x) * 0.001;
      this.vy += (cohesionY - this.y) * 0.001;
      this.vx += separationX * 0.02;
      this.vy += separationY * 0.02;
    }

    const globalPull = 0.00008;
    this.vx += (width / 2 - this.x) * globalPull;
    this.vy += (height / 2 - this.y) * globalPull;

    const speedSq = this.vx * this.vx + this.vy * this.vy;
    const maxSpeed = 4;
    if (speedSq > maxSpeed * maxSpeed) {
      const scale = maxSpeed / Math.sqrt(speedSq);
      this.vx *= scale;
      this.vy *= scale;
    }

    this.x += this.vx;
    this.y += this.vy;
  }

  draw() {
    ctx.fillRect(this.x | 0, this.y | 0, 1, 1);
  }
}

for (let i = 0; i < BOID_COUNT; i++) {
  boids.push(new Boid());
}

function animate() {
  ctx.fillStyle = "black";
  ctx.fillRect(0, 0, width, height);
  ctx.fillStyle = "white";

  frame++;
  updateFields();
  buildGrid();

  for (let i = 0; i < boids.length; i++) {
    if (i % 4 === frame % 4) boids[i].update();
    boids[i].draw();
  }

  time += 0.003;
  requestAnimationFrame(animate);
}

animate();

const CONFIG = {
    headerImage: "https://redesencontros.github.io/dancaeissoaquio/assets/header.png",
    buttonImage: "https://redesencontros.github.io/dancaeissoaquio/assets/button.png",
    buttonSound: "https://redesencontros.github.io/dancaeissoaquio/assets/button-sound.mp3",
    contentListTxt: "https://redesencontros.github.io/dancaeissoaquio/content-list.txt"
};

let deck = [];
let currentIndex = 0;
let currentAudio = null;
let buttonClickSound = new Audio(CONFIG.buttonSound);

document.getElementById("headerImage").src = CONFIG.headerImage;
document.getElementById("buttonImage").src = CONFIG.buttonImage;

fetch(CONFIG.contentListTxt)
    .then(res => res.text())
    .then(text => {
        deck = text.split("\n").map(l => l.trim()).filter(l => l);
        shuffle(deck);
    });

function shuffle(array) {
    for (let i = array.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        [array[i], array[j]] = [array[j], array[i]];
    }
}

document.getElementById("gameButton").addEventListener("click", () => {

    buttonClickSound.currentTime = 0;
    buttonClickSound.play();

    if (currentAudio) {
        currentAudio.pause();
        currentAudio = null;
    }

    if (currentIndex >= deck.length) {
        shuffle(deck);
        currentIndex = 0;
    }

    showContent(deck[currentIndex++]);
});

function showContent(url) {
    const display = document.getElementById("contentDisplay");
    display.innerHTML = "";

    const lower = url.toLowerCase();

    if (lower.match(/\.(jpg|jpeg|png|gif|webp)$/)) {
        const img = document.createElement("img");
        img.src = url;
        display.appendChild(img);
    }

    else if (lower.match(/\.(mp4|webm)$/)) {
        const video = document.createElement("video");
        video.src = url;
        video.autoplay = true;
        video.loop = true;
        video.muted = true;
        video.playsInline = true;
        display.appendChild(video);
    }

    else if (lower.match(/\.(mp3|wav|ogg)$/)) {
        const audio = new Audio(url);
        audio.loop = true;
        audio.play();
        currentAudio = audio;
    }

    else if (lower.match(/\.txt$/)) {
        fetch(url)
            .then(res => res.text())
            .then(text => {
                const div = document.createElement("div");
                div.style.color = "white";
                div.style.textAlign = "center";
                div.style.padding = "20px";
                div.innerText = text;
                display.appendChild(div);
            });
    }
}
